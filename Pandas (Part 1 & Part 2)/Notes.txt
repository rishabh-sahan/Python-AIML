Pandas -> Cleaning Data
    Handle Missing Values
        isnull() / isna() <- alias
        isnull().sum()
        dropna()
        fillna(value)
        ffill() ->  forward fill
        bfill() ->  backward fill
    Handle Duplicate Values
        duplicated()
        drop_duplicates()
    Handle Data Types & data-time
        dtypes
        astype(newType)
        to_datetime()
    Handle Strings
        .str.lower(), .str.upper() & .capitalize()
        .str.strip()
        .str.split("")
        .str.contains(val, case=False)
